# PL/0子集语言的词法规则

# 关键字（这些必须在ID之前检查）
VAR = var
WHILE = while
IF = if
ELSE = else
READ = read
WRITE = write

# 逻辑运算符
AND = and
OR = or
NOT = not

# 标识符（在关键字之后检查）
ID = [a-zA-Z_][a-zA-Z0-9_]*

# 数字
NUM = [0-9]+

# 运算符
ASSIGN = =
PLUS = \+
MINUS = -
MUL = \*
DIV = /
MOD = %

# 比较运算符
LT = <
LE = <=
GT = >
GE = >=
EQ = ==
NE = <>

# 括号和分隔符
LPAREN = \(
RPAREN = \)
LBRACE = \{
RBRACE = \}
SEMI = ;
COMMA = ,
