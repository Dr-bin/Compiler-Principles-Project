# PL/0子集语言的词法规则

# 关键字
CONST = const
VAR = var
PROCEDURE = procedure
CALL = call
WHILE = while
IF = if
ELSE = else
READ = read
WRITE = write
ODD = odd
BEGIN = begin
END = end

# 逻辑运算符
AND = and
OR = or
NOT = not

# 标识符
ID = [a-zA-Z_][a-zA-Z0-9_]*

# 数字
NUM = [0-9]+

# 运算符
ASSIGN = =
PLUS = \+
MINUS = -
MUL = \*
DIV = /
MOD = %

# 比较运算符
LT = <
LE = <=
GT = >
GE = >=
EQ = ==
NE = <>

# 括号和分隔符
LPAREN = \(
RPAREN = \)
LBRACE = \{
RBRACE = \}
SEMI = ;
COMMA = ,
